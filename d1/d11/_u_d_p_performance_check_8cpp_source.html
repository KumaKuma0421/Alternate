233 _tprintf (_T ("UdpController::Start(IP:%s, Port:%d, kickoff:%d) IN.\n"), lpctszIP, wPort, kickOff);

235 _connector = _builder->CreateUdpConnector (lpctszIP, wPort);

274DWORD UdpController::Invoke (LPVOID lpvParam)
275{
276 _tprintf (_T ("UdpController::Invoke(LPVOID) IN.\n"));
277
278 BYTE byData[DATA_SIZE];
279
280 int count = 0;
281 while (_stop == false)
282 {
283 INT recvSize = _connector->Recv (byData, sizeof (byData));
284 if (recvSize <= 0)
285 {
286 _tprintf (_T ("UdpController::Invoke(LPVOID)\n"));
287 _tprintf (_T (" recvSize=%d Socket disconnected. reason:%d\n"),
288 recvSize, WSAGetLastError ());
289 return -1;
290 }
291#ifdef _DUMP_DATA
292 DumpData (_T ("UDP"), 0, byData, recvSize);
293#else
294 _tprintf (_T ("%d\r"), count++);
295#endif
296 int dataLen = MakeData (byData);
297
298 INT sendSize = _connector->Send (byData, dataLen);
299 if (sendSize != dataLen)
300 {
301 _tprintf (_T ("UdpController::Invoke(LPVOID)\n"));
302 _tprintf (_T (" send=%d sended=%d Invalid send size. reason:%d\n"),
303 dataLen, sendSize, WSAGetLastError ());
304 return -2;
305 }
306 }
307
308 _tprintf (_T ("UdpController::Invoke(LPVOID) OUT.\n"));
309 return 0;
